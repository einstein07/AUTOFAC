474 static std::string mapToString(const map_t& m, const std::string& prefix = std::string("")) {
475 std::stringstream ofs;
476
477 size_t behav_dim = m.dimensionality;
478 size_t offset = 0;
479 for (const Elite* i = m.data(); i < (m.data() + m.num_elements()); ++i) {
480 if (i) {
481 behav_index_t index;
482 for (unsigned int dim = 0; dim < behav_dim; dim++ ) {
483 index[dim] = (offset / m.strides()[dim] % m.shape()[dim] + m.index_bases()[dim]);
484 }
485
486 assert(m(index).fitness_ == (i)->fitness_);
487
488 ofs << prefix << offset << ",";
489 for (size_t dim = 0; dim < behav_dim; ++dim) {
490 ofs << index[dim] / (float)m.shape()[dim] << ",";
491 }
492 ofs << m(index).fitness_;
493 ofs << std::endl;
494 }
495 ++offset;
496 }
497
498 return ofs.str();
499 }
